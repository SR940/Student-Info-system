It's a Java Netbeans Project. I've used Oracle Database. When you run this project, at login page you need user id and password to log in. 
So, you need to create a database atfirst. Here is all the data tables.

CREATE TABLE  "STUDENT" 
   (	"ID" VARCHAR2(30), 
	"F_NAME" CHAR(25), 
	"LNAME" CHAR(15), 
	"GENDER" CHAR(30), 
	"FNAME" VARCHAR2(30), 
	"MNAME" VARCHAR2(30), 
	"DOB" VARCHAR2(30), 
	"CONT" NUMBER(20,0), 
	"EMAIL" VARCHAR2(100), 
	"ADDRESS" VARCHAR2(200), 
	"DEPT" CHAR(10), 
	 PRIMARY KEY ("ID") ENABLE
   )

CREATE TABLE  "ADMIN" 
   (	"ID" VARCHAR2(30) NOT NULL ENABLE, 
	"NAME" VARCHAR2(30) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(20), 
	 PRIMARY KEY ("ID") ENABLE
   )

CREATE TABLE  "COURSES" 
   (	"TRIMESTER" VARCHAR2(30), 
	"COURSE_CODE" VARCHAR2(15), 
	"COURSE_TITLE" VARCHAR2(150) NOT NULL ENABLE, 
	"CREDIT_HR" VARCHAR2(15) NOT NULL ENABLE, 
	 PRIMARY KEY ("COURSE_CODE") ENABLE, 
	 FOREIGN KEY ("TRIMESTER")
	  REFERENCES  "TRIMESTER" ("TRIMESTER") ENABLE
   )

CREATE TABLE  "ST_COURSE" 
   (	"ID" VARCHAR2(30), 
	"TRIMESTER" VARCHAR2(30), 
	"COURSE_CODE" VARCHAR2(15), 
	"COURSE_TITLE" VARCHAR2(150) NOT NULL ENABLE, 
	"CREDIT_HR" VARCHAR2(15) NOT NULL ENABLE, 
	 FOREIGN KEY ("TRIMESTER")
	  REFERENCES  "TRIMESTER" ("TRIMESTER") ENABLE, 
	 FOREIGN KEY ("COURSE_CODE")
	  REFERENCES  "COURSES" ("COURSE_CODE") ENABLE, 
	 FOREIGN KEY ("ID")
	  REFERENCES  "STUDENT" ("ID") ENABLE
   )

CREATE TABLE  "TRIMESTER" 
   (	"TRIMESTER" VARCHAR2(30), 
	 PRIMARY KEY ("TRIMESTER") ENABLE
   )


CREATE TABLE  "TUITION" 
   (	"ID" VARCHAR2(30), 
	"P_DATE" VARCHAR2(30), 
	"T_FEE" NUMBER(10,3), 
	 FOREIGN KEY ("ID")
	  REFERENCES  "STUDENT" ("ID") ENABLE
   )
